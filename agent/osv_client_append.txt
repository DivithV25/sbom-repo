
    # Cache the results
    if use_cache:
        _osv_cache.set(package_name, version, vulnerabilities, ecosystem)

    return vulnerabilities

    return _cache_manager.get_stats()

    """"""Clear all OSV cache entries""""""
    return _osv_cache.clear()
